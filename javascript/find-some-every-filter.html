数组中常用的方法
// find的基本用法 let array = [1,2,3,4,5] // 判断数组是否有小于3的元素 array.find(item=>item<3) // 返回第一个符合条件的元素 1
// some的基本使用 let array = [1,2,3,4,5] // 判断数组是否有小于3的元素 array.some(item=>item<3) // 返回true或false true
// every的基本使用 let array = [1,2,3,4,5] // 判断数组元素是否全部小于3 array.every(item=>item<3) // 返回true或false false
// filter的基本使用 let array = [1,2,3,4,5] // 判断数组有小于3的全部元素 array.filter(item=>item<3) // 返回小于3的数组 [1,2]
