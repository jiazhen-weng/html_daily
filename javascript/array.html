数组的增删改合并
// 数组添加元素 let array = [1,2,3,4,5] array.push(6) array.push(7) [1,2,3,4,5,6,7]
// 数组删除元素 let array = [1,2,3,4,5] // 删除元素3 array.splice(2,1) [1,2,4,5]
// 数组修改元素 let array = [1,2,3,4,5] // 把最后一个元素修改为100 let last = array.length-1 array[last]=100 [1,2,3,4,100]
// 两个数组合并 let one = [1,2,3] let two = [8,9,10] [...one,...two] [1,2,3,8,9,10]
